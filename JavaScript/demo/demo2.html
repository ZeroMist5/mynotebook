3.到controller.js中写添加英雄信息的具体业务逻辑，代码如下 3 4 5 18-----32

// 此模块是专门用来处理业务逻辑的
// 这些里面都是些功能方法
const fs = require('fs')
const path = require('path')
const moment = require('moment')
let arr = require('./heros.json')
module.exports = {
  // 1. 渲染主页面
  renderIndex(req,res){
    // let arr = require('./heros.json')
    res.render('index',{arr})
  },
  // 2. 显示添加英雄的页面
  showAddHero(req,res){
    res.render('add',{})
  },
  // 3. 添加英雄信息
  addHeroInfo(req,res){
    // 3.1 接收post方式传递过来的数据
    let hero = req.body
    // 3.2 添加id 还是使用时间戳
    hero.id = Date.now() + Math.trunc(Math.random()*1000000)
    // 3.3 添加创建时间
    hero.date = moment().format('YYYY-MM-DD HH:mm:ss')
    // 3.4 将英雄信息添加到数组中
    arr.push(hero)
    // 3.5 将数组再写回heros.json文件中
    fs.writeFile(path.join(__dirname,'./heros.json'),JSON.stringify(arr,null,' '),err=>{
      if(err) return res.send('添加英雄失败')
      // 3.6 如果添加成功了 要跳转到主页面
      res.redirect('/')
    })
  }
}

// 5. 删除一条英雄数据
  delHeroInfo(req,res){
    // 5.1 接收传递过来的id 根据id删除数据
    let {id} = req.query
    // 5.2 根据id查找到英雄数据
    let index = arr.findIndex(item=> item.id == id)
    // 5.3 删除英雄数据
    arr.splice(index,1)
    // 5.4 写回heros.json文件中
    fs.writeFile(path.join(__dirname,'./heros.json'),JSON.stringify(arr,null,' '),err=>{
      if(err) return res.send('删除英雄失败')
      // 5.5 删除成功要重新刷新主页面 就是重定向
      res.redirect('/')
    })
  }

// 7. 更新英雄信息
  updateHeroInfo(req,res){
    // 7.1 获取传递过来的数据
    let hero = req.body
    // 7.2 获取POST过来的数据中的id
