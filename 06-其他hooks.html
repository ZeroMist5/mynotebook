06. 其它hooks

React18之前的HOOK

useContext

作用：获取context数据的第三种写法，前面两种分别是Provider、Consumer，contextType。 用法：const context = useContext(使用createContext()接受的那个变量名所在的组件)

// Test.js
import React,{useContext} from 'react'
import testContext from './context'
export default function Test() {
  const context = useContext(testContext)
  console.log(context)
  return <div></div>
}

useReducer

作用：和redux完全没有关系，和useState作用是一样的。useSate有的缺点就是，有多少个数据就需要调用多少次useState，而useReducer不用，但是对比useState更繁琐

import React, { useReducer } from 'react'

const initialState = { count: 0, msg: '哈哈' }

function reducer(state, action) {
  switch (action.type) {
    case 'increment':
      return {
        ...state,
        count: state.count + 1,
      }
    case 'decrement':
      return {
        ...state,
        count: state.count - 1,
      }
    default:
      throw new Error()
  }
}
export default function Test1() {
  const [state, dispatch] = useReducer(reducer, initialState)
  return (
    <>
      Count: {state.count}
      msg: {state.msg}
      <button onClick={() => dispatch({ type: 'decrement' })}>-</button>
      <button onClick={() => dispatch({ type: 'increment' })}>+</button>
    </>
  )
}

// useReduder的第三个参数:

function init(initialCount) {
  return {count: initialCount};
}

function reducer(state, action) {
  switch (action.type) {
    case 'increment':
      return {count: state.count + 1};
    case 'decrement':
      return {count: state.count - 1};
    case 'reset':
      return init(action.payload);
    default:
      throw new Error();
  }
}

function Counter({initialCount}) {
  const [state, dispatch] = useReducer(reducer, initialCount, init);
  return (
    <>
      Count: {state.count}
      <button
        onClick={() => dispatch({type: 'reset', payload: initialCount})}>
        Reset
      </button>
      <button onClick={() => dispatch({type: 'decrement'})}>-</button>
      <button onClick={() => dispatch({type: 'increment'})}>+</button>
    </>
  );
}

useCallback

​ 首先我们先看下面代码的场景：

import React,{useState} from 'react'
export default function Test() {
  console.log('test渲染了')
  const [count,setCount] = useState(0)
  const handle = () => {
    console.log(count)
    setCount(count + 1)
  }
  return (
    <div>
      <div>{count}</div>
      <button onClick={handle}>按钮</button>
    </div>
  )
}

​ 当每次单击button按钮的时候，handle函数则会调用，从而渲染数据，每次点击的时候handle函数都意味着会重新创建一次，这样性能不好所以我们可以使用useCallback这个hook来优化这种代码

import React,{useState} from 'react'
export default function Test() {
  console.log('test渲染了')
  const [count,setCount] = useState(0)
  // 这样下去就可以让这个函数只创建一次
  const handle = useCallback(() => {
    console.log(count)
    setCount(count + 1)
  },[])
  return (
    <div>
      <div>{count}</div>
      <button onClick={handle}>按钮</button>
    </div>
  )
}

由于上面的count会形成闭包，加上useCallback会缓存这个函数，所以每次执行的都是上一次闭包的，这就导致了，count + 1之后，button按钮无论单击多少次，都只是显示1

如果在[]里面写count，这样的确可以解决问题，但是依然button按钮依然会随着count的变化而重新创建函数

可以在setCount中写一个回调函数

setCount((count) => { return count + 1 })

useMemo

useCallback 可以缓存并返回一个函数，减少函数的创建次数 useMemo 可以缓存并返回一个函数的计算结果，减少函数的执行次数

​ 首先看下面的代码

import React, { useState, useMemo } from 'react'

export default function Test() {
  const [count, setCount] = useState(1)
  const [val, setValue] = useState('')

  function expensive() {
    console.log('compute')
    let sum = 0
    for (let i = 0; i < count * 100; i++) {
      sum += i
    }
    return sum
  }


  return (
    <div>
      <h4>
        {count}-{val}-{expensive}
      </h4>
      <div>
        <button onClick={() => setCount(count + 1)}>+c1</button>
        <input value={val} onChange={(event) => setValue(event.target.value)} />
      </div>
    </div>
  )
}

​ 当输入框在输入的时候因为改变了val的值，会导致expensive多次调用，这样是不合理的。所以可以使用useMemo这个hook，使用用发和useCallback一样,但是useCallback返回的还是那个函数，而useMemo是返回这个函数的返回值

const expensive = useMemo(function () { console.log('compute') let sum = 0 for (let i = 0; i < count * 100; i++) { sum += i } return sum },[count])

useCallback和useMemo的区别

1、useCallback返回的是一个函数，useMemo返回的是计算之后的结果 2、useCallback缓存的是函数，useMemo缓存的是计算之后的结果

React.Memo

React.memo不是hook 作用：判断父组件传入的props数据是否发生变化，如果变化则更新，如果不变则不更新

import React from 'react'
  function Test() {
    console.log('test更新了')
    return <div>Test</div>
  }
  // 实际上还是Test组件，只是在外面包了一层，加了判断
  // 在使用的时候Test组件需要接受一个值，来根据这个值是否变化来更新Test组件
  // <Test prpos={this.state.count}></Test>
}
  export default React.memo(Test)

React.memo、useCallback、useMemo的区别？

1、React.memo不是hook，useCallback、useMemo是hook

2、作用：

​ React.memo的作用是：判断父组件传入的props数据是否发生变化，如果变化则更新，如果不变则不更新 ​ useCallback的作用：可以缓存并返回一个函数，减少函数的创建次数 ​ useMemo的作用：useMemo 可以缓存并返回一个函数的计算结果，减少函数的执行次数

创建ref对象的新方法

import React,{useRef} from 'react'

export default function App() {
  const ref = useRef()
  return (
    <div ref={ref}>

       console.log(ref)

useImperativeHandle

useImperativeHandle可以让你在使用ref时自定义暴露给父组件的实例值。useImperativeHandle应当与forwardRef一起使用 作用：封装公共组件的时候，可以给使用公共组件的组件提供指定的API，有条件的操作公共组件的真实dom

比如在App组件中，有Test组件，在App组件中需要拿到Test组件中的真实dom，因为Test是别人封装的公共组件，你只能向封装Test组件的作者给一个方法来获取，那个作者会使用forwardRef

// App.js
import React,{useRef} from 'react'
import Test from './Test'
export default function App() {
   const appRef = useRef()
   return (
     <div>
       App
       <button onClick = {() => {
       // 调用聚焦
       appRef.current.xxx()
       // appRef.current.focus()
       // 能够修改公共组件的样式，这是不合理的
       // appRef.current.style.backgroundColor = 'pink'
   }}>点击按钮让Test组件中input获取焦点</button>
       <Test ref={appRef}></Test>
     </div>
   )
}

// Test.js
import React ,{useImperativeHandle,useRef} from 'react'

export default React.forwardRef(function Test(props,appRef) {
    const testRef = useRef()
    // 使用这个hook可以自定义ref对象current的值
    useImperativeHandle(appRef,() => {
        // 这个回调返回值是什么，则appRef的current指向的就是什么
        return {
            xxx () {
                testRef.current.focus()
            }
        }
    })
  return (
    <div>
      Test
      <input type="text" ref={testRef}/>
    </div>
  )
})

useLayoutEffect

useLayoutEffect和useEffect作用是一样的，也是模拟那三个生命周期钩子函数,但是执行时机略有不同 useEffect挂载一定是页面渲染完毕了再执行回调函数，不会阻塞页面渲染，渲染完了才执行 useLayoutEffect会阻塞页面渲染，等回调执行函数执行完页面才渲染

React18新增的HOOK

背景：React18之前，React渲染页面的时候是线性的。如果更新了两个状态的值，例如：setX、setY，先渲染前者，没有渲染完，则不会渲染后面的。到了React18开始，渲染是同时的，setX,setY都是同时渲染。如果使用的设备硬件太差，同时渲染会导致页面卡顿

useTransition

const [isPending,startTransition] = useTransition() isPending 是一个布尔值，如果紧急任务正在渲染就是true，不是就是false，返回的是紧急任务 startTransition (() => { 写需要设置非紧急任务的函数调用 })

useDeferredValue

和useTransition差不多 用法：const result = useDeferredValue(值)，返回一个延迟的状态值，React内部会让这个值得相关渲染慢一点

useId

作用：返回一个唯一的id值 import React , { useId } from 'react' export default function App () { const id1 = useId() const id2 = useId() const id3 = useId() }
