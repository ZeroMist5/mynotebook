09-JavaScript常用的内置函数

函数的分类

  • 内置函数：即js内置就有的，可以直接使用。
  • 自定义函数：程序员自己封装的函数。

可变和不可变函数

  • 若某个方法不会影响原变量，这些方法称之为不可变。
  • 若某个方法会影响原变量的，这些方法称之为可变的。

后面字符串相关方法都是不可变的。 而数组方法部分可变部分不可变。

字符串常用内置函数

既然是字符串的相关方法，即只能是字符串才能调用。

// 若某个方法不会影响原变量，这些方法称之为不可变。 如：字符串方法基本都是不可变的。
// 若某个方法会影响原变量的，这些方法称之为可变的。 如：数组方法部分可变部分不可变。
// 1. str.charAt(index): // 获取指定下标得到字符
console.log(  'abcdef'.charAt() ); // 'a'
console.log(  'abcdef'.charAt(2) ); // 'c'
console.log(  'abcdef'.charAt(20) ); // ''

var str = 'abc'
console.log(  str[2] ); // 'c'


// 建议使用charAt方法，因为兼容性更好，可以兼容到IE

// 2. toLowerCase()：对字符串统一转为小写
// 2. toUpperCase()：对字符串统一转为大写
var str = 'Abc';
console.log( str.toLowerCase() )
console.log( str.toUpperCase() )


// 3. str.substring(indexStart[, indexEnd]): 截取指定下标范围的子串。不含最后末尾下标，
// 若末尾下标不写，则截取到字符串的结尾。
console.log('abcdef'.substring(0,3)); //'abc' 
console.log('abcdef'.substring(2,4)); //'cd' 
console.log('abcdef'.substring(2)); //'cdef' 


// 4. indexOf(substr); 返回子串所在目标字符串中第一个字符的下标，没找到则返回 -1;
console.log( 'abcdbef'.indexOf('b') ); // 1
console.log( 'abcdbef'.indexOf('e') ); // 5
console.log( 'abcdbef'.indexOf('l') ); // -1
console.log( 'abcdbef'.indexOf('a') ); // 0
console.log( 'abcdbef'.indexOf('cdb') ); // 2

// 5. str.includes(searchString[, position]): 用来判断目标字符串中是否含有特定的子串，有返回true,否则返回false
console.log(  'abcdbef'.includes('b') ); // true
console.log(  'abcdbef'.includes('abc') ); // true
console.log(  'abcdbef'.includes('abc',4) ); // false
console.log(  'abcdbef'.includes('lmn') ); // false
console.log(  'abcdbef'.includes('a') ); // true
console.log(  'abcdbef'.includes('a',2) ); // false

// 6. str.replace(substr); // 对目标字符串进行替换，返回一个新的字符串
var str = 'hello ok';
console.log(str.replace('hello','hi')); // 'hi ok'
console.log('i like apple'.replace('apple','orange')); // 'hi ok'

// 7. str.repeat(count): // 重复一个字符串多少次。
console.log("*".repeat(4)); // '****'
console.log("abc".repeat(4)); // 'abcabcabcabc'

// 8. str.trim(): // 去除字符串两端的连续的空格
//   str.trimStart(); // 去除字符串开头的连续的空格
//   str.trimEnd(); // 去除字符串结尾的连续的空格

var title = "  I like you     ";
console.log(title)
console.log( title.trim() );
console.log( title.trimStart() );
console.log( title.trimEnd() );


// 9. str.split([separator[, limit]]); // 将一个字符串用特点字符进行分割,返回到一个新数组
console.log( 'abcbdf'.split() ); // ['abcdef']
console.log( 'abcbdf'.split('') ); // [a,b,c,b,d,f]
console.log( 'ab-cb-df'.split('-') ); // [ab,cb,df]
console.log( 'ab-cb-df'.split('-',1) ); // [ab]


// 10. str.startsWith(searchString[, position]); // 是否以指定子串开头。
// . str.endsWith(searchString[, position]); // 是否以指定子串结尾。
console.log('v-model'.startsWith('v-')); // true
console.log('model'.startsWith('v-')); // false
console.log('model-v'.endsWith('-v')); // true

数组常用内置函数

  • 可变的函数(7个)： push、pop、unshift、shift、reverse，splice, sort(排序)
  • 不可变的函数： concat、includes、join、indexOf,slice

可变函数：



        var arr = [1,2,3];
        console.log( arr.push(4),arr); // push尾部添加
        console.log( arr.unshift(5),arr); // unshift头部添加
        console.log( arr.pop(),arr); // pop,弹出尾部最后的元素
        console.log( arr.shift(),arr); // shift,弹出头部最后的元素
        console.log( arr.reverse(),arr ) // 翻转数组

        // splice：1.删除元素 2.删除并替换元素
        var arr2 = ['蔡特','张先锐','叶浩坚','陈安烂'];
        // 从下标1开始删除，删1个
        // console.log( arr2.splice(1,1),arr2 )

        // 从下标1开始删除，删1个，并替换成苏春连
        console.log( arr2.splice(1,1,'苏春连'),arr2 )

不可变：

console.log( arr2.includes('烂哥') ); // true

console.log( arr2.indexOf('烂哥') ); // 1 


console.log(arr1.join(',')); // join()用特定字符将数组元素链接在一起，并返回连接后的字符串。


// Array.isArray()： 判断变量是否是数组

内置Date时间对象的常用方法

通过Date构造函数可以创建一个时间对象，用来解决时间相关的问题。

// 基于当前时间创建时间对象
let date = new Date()
// 基于指定时间创建时间对象
// let date = new Date('2018-08-08 12:23:45')

console.log(date);
// 获取年月日、时分秒、毫秒
console.log(date.getFullYear());
console.log(date.getMonth() + 1);
console.log(date.getDate());
console.log(date.getHours());
console.log(date.getMinutes());
console.log(date.getSeconds());
console.log( date.getMilliseconds() );

// 即自 1970 年 1 月 1 日（UTC）起经过的毫秒数
// 1s = 1000ms
console.log(Date.now()); // 毫秒值
console.log(parseInt(Date.now() / 1000)); // 秒值

案例：封装一个函数，可以得到当前日期时间，返回格式如 2018-08-07 12:23:34

function getNowDate() {
  let date = new Date();
  let year = date.getFullYear();
  let month = date.getMonth() + 1;
  let day = date.getDate();
  let hour = date.getHours();
  let minute = date.getMinutes();
  let second = date.getSeconds();
  // 补0函数
  function fillZero(number){
    number = number.toString();
    return number >= 10 ? number : '0' + number;
  }
  return year + '-' + fillZero(month) + '-' + fillZero(day) + ' ' + fillZero(hour) + ':' + fillZero(minute) + ':' + fillZero(second)

}
console.log( getNowDate() )

    return Math.ceil( Math.random() * n )

  • 封装一个函数，生成指定范围的随机数

    function randomInt(min, max) {

}

数组排序方法sort原理

  • 如果不给sort传递参数：此时排序的方式是按照ascii码进行排序 
var arr = [3,11,21,24,1]   
arr.sort()
//此时原数组已经被改变
console.log(arr) // [1, 11, 21, 24, 3]

ASCII 码表：一共规定了128个字符的编码，比如空格SPACE是32（二进制00100000），大写的字母A是65（二进制01000001）。小写字母a是97，b是98，c是99， 以此类推

sort实现数组元素升序或降序：

  • 升序： 第一个元素 大于 第二个元素
  • 降序：第二个元素 大于 第1个元素
var arr = [3,11,21,24,1]

arr.sort(function(a,b){
    return  a - b; // 升序 
    // return  b - a; // 降序
})

console.log(arr); // [1, 3, 11, 21, 24]

获取字符Ascill编码的值：

"abc".charCodeAt(); // 97 ， 默认下标为0
"abc".charCodeAt(1); // 98  
"abc".charCodeAt(2); // 99 

'1'.charCodeAt(); // 49
'2'.charCodeAt(); // 50

字母和字母之间的比较是按照其ascill码表进行比较的，如果第一个字母ascill码一样，则继续比较后面的字符

'a' > 'b' // false
'ab' > 'ac' // false
'ac' > 'ab' // true

'3' > '22' // true
3 > 22 // false

思维提升：计算n-m之间的和

  • 普通程序员
let total = 0;
console.time('time1')
for(let i = 1; i <= 100000000; i++ ){
  total += i;
}
console.timeEnd('time1')
console.log(total)
  • 高程程序员
// 不利用循环计算 1+2+。。。+n的和

// 1 + 2 + ....+n -1 + n =  n + n - 1....+ 2 + 1

1 + 2 + 3 = 3 + 2 + 1

1 + 2 + 3
3 + 2 + 1

// 归纳 ( min + max * (max - min + 1) ) / 2



// 求出任意两个之间的总和
function sum(m,n){
  // 第一项1 + 最后一项n , 共 （n个数）最后结果除以2
  return  ( (m + n) * (n-m + 1) ) / 2
}

console.time('time2')
console.log( sum(1,100) ) 
console.timeEnd('time2')
