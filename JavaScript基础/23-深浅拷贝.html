23-深浅拷贝

深浅拷贝（克隆）

回顾基本类型和引用类型赋值

  • 基本类型：按值传递，修改新的不会影响旧的
  • 引用类型：按址传递，修改新的会影响旧的

基本类型：

var a = 1;
var b = a; // 把值复制一份给b
b = 2;
console.log(a);// 1
console.log(b);// 2

引用类型：

var obj1 = {
    name:'张三'
}
var obj2 = obj1 // 引用赋值
obj2.name = '李四'
console.log(obj1.name); // 李四
console.log(obj2.name); // 李四

解释：

上面的对象值都是李四，说明obj1和obj2是共享同一块内存，所以修改新的会影响旧的。

我们期望修改新的不会影响旧的对象！！，怎么解决？？ 通过深拷贝即可

什么是深浅拷贝

浅拷贝：浅拷贝只复制指向某个对象的引用，而不复制对象本身，新旧对象还是共享同一块内存。修改新对象会影响原对象。

深拷贝：深拷贝会创造一个一模一样的对象，新对象跟原对象不共享内存，修改新对象不会影响到原对象

手动实现深拷贝

想要实现深拷贝，必须重新开辟一个新的内存空间去存储新对象，这样修改新的就不会影响旧的。

function cloneDeep(target){
    // 只对对象进行深拷贝，不是对象类型则直接原样返回
    if(typeof target !== 'object'){
        return target
    }
    var obj = {}; // 创建新对象，新内存空间
    for(var k in target){
        obj[k] = target[k]
    }
    return obj;
}

var obj1 = {name:'张三'}
var obj2 = cloneDeep(obj1)

obj2.name = '李四'
console.log(obj1.name); // 张三
console.log(obj2.name); // 李四

上面的深拷贝方法只能实现对对象进行深拷贝，但数组也是object类型，所以我们还要考虑数组的情况：

function cloneDeep(target) {
  // 只对对象进行深拷贝，不是对象类型则直接原样返回
  if (typeof target !== 'object') {
    return target
  }

  var type = Object.prototype.toString.call(target)
  var result = type === '[object Object]' ? {} : [] // 创建新对象，新内存空间
  if (type === '[object Object]') {
    for (var k in target) {
      result[k] = target[k]
    }
  } else if (type === '[object Array]') {
    for (var i = 0; i < target.length; i++) {
      result[i].push(target[i])
    }
  }

  return result
}

// 数组情况：
var arr1 = [1, 2, 3]
var arr2 = cloneDeep(arr1)
arr2.push(4)
console.log(arr1) // [ 1, 2, 3]
console.log(arr2) // [ 1, 2, 3, 4 ]

// 对象情况：
var obj1 = { name: '张三' }
var obj2 = cloneDeep(obj1)

obj2.name = '李四'
console.log(obj1.name) // 张三
console.log(obj2.name) // 李四

上面拷贝对象和数组都没问题，但仅能拷贝一层，如果是多层，且值是对象的话，依然是引用传值。

// 对象嵌套多层情况：
var obj1 = { name: '张三', hobby: ['篮球', '台球'] }
var obj2 = cloneDeep(obj1)

obj2.name = '李四'
console.log(obj1.name) // 张三
console.log(obj2.name) // 李四

console.log(obj1.hobby === obj2.hobby) // true 依然是指向同一块内存

obj2.hobby.push('羽毛球')
console.log(obj1.hobby) // [ '篮球', '台球', '羽毛球' ]
console.log(obj2.hobby) // [ '篮球', '台球', '羽毛球' ]

可见修改name没问题，修改hobby就有问题了。

如何解决：递归！

// 实现深拷贝函数
function cloneDeep(target) {
    // 如果传入的参数不是引用类型，则直接返回
    if (typeof target !== 'object') {
        return target
    }
    // 判断是数组还是对象
    var newObj = target.constructor === Array ? [] : {}
    var type = Object.prototype.toString.call(target)

    if (type === '[object Object]') {
        // 对象
        for (var k in target) {
            console.log(target[k]) // 欣姐,['化妆'] 可能是基本或引用类型
            newObj[k] = cloneDeep(target[k]) // 递归调用
        }
    } else {
        // 数组
        for (var i = 0; i < target.length; i++) {
            newObj.push(cloneDeep(target[i])) // 递归调用
        }
    }

    return newObj
}

// 对象嵌套多层情况：
var obj1 = { name: '张三', hobby: ['篮球', '台球'] }
var obj2 = cloneDeep(obj1)

console.log(obj1.hobby === obj2.hobby) // false

obj2.hobby.push('羽毛球')
console.log(obj1.hobby) // [ '篮球', '台球']
console.log(obj2.hobby) // [ '篮球', '台球', '羽毛球' ]

其他深拷贝方式

  • 利用lodash工具库的_.cloneDeep(value) 实现深拷贝

  • 内置函数JSON.stringify(对象)实现深拷贝。但是有缺点！！！

注意 ：通过JSON.stringify()可以对简单的对象进行深拷贝，最后再用JSON.parse(字符串)函数转回来即可。

但是它有缺点，一些特殊的值会丢失，如：

  • 对于属性值类型是undefined、function、symbol等类型，会丢失对应值选项

  • 对于属性值类型是err、regexp 这些类型，属性值会变为空对象。

  • 对于属性值类型是date等类型，属性值会变为字符串。即使再次转回对象，属性值还是字符串
  • ...
var obj1 = { name: '悦姐', b: 1, a: undefined, getName: function () {} }
// 将 对象 =》 字符串
var str = JSON.stringify(obj1)
console.log(str) // '{"name":"悦姐"}'

// 将字符串 => 对象
var obj2 = JSON.parse(str)

console.log(obj1 === obj2)
obj2.name = '瑞姐'
console.log(obj1) // {name: '悦姐', b: 1, a: undefined, getName: ƒ}
console.log(obj2) // {name: '瑞姐', b: 1}
